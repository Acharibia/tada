A lightweight, customizable toast notification library

Toast Types

tada.success("Operation successful!");
tada.error("Something went wrong!");
tada.warning("Be careful!");
tada.info("Did you know?");
tada.show("Default message");

Positions

tada.show("Message", { position: "top-left" });
tada.show("Message", { position: "bottom-center" });

Advanced Features

// Long duration
tada.show("Message", { duration: 10000 });

// No auto close
tada.show("Message", { duration: 0 });

// Custom animation
tada.show("Message", {
  animation: {
    enter: "scale(0)",
    exit: "scale(0)",
    duration: 500
  }
});

// Max toasts limit
const toast = new TadaToast({ maxToasts: 3 });

// Clear all toasts
tada.clear();
